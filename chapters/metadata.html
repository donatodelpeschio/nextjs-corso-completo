La Tua Domanda Iniziale! 🎯

Ecco come risolvere il problema del canonical e meta tags per il tuo blog in Next.js 15+!

Metadata API Statica

Il modo più semplice - esporta un oggetto metadata:

// app/page.js
export const metadata = {
  title: 'Il Mio Blog',
  description: 'Blog su web development',
  keywords: ['nextjs', 'react', 'javascript'],
  authors: [{ name: 'Tuo Nome' }],
  openGraph: {
    title: 'Il Mio Blog',
    description: 'Blog su web development',
    images: ['/og-image.jpg'],
  }
}

export default function Home() {
  return <h1>Homepage</h1>
}

Metadata Dinamica per Blog

Per pagine dinamiche, usa generateMetadata:

// app/blog/[slug]/page.js

export async function generateMetadata({ params }) {
  const { slug } = await params  // ✅ Next.js 15+!
  
  const post = await fetch(`https://api.example.com/posts/${slug}`)
    .then(res => res.json())
  
  return {
    title: post.title,
    description: post.excerpt,
    
    // 🎯 CANONICAL - ecco come si fa!
    alternates: {
      canonical: `/blog/${slug}`
    },
    
    // Open Graph
    openGraph: {
      title: post.title,
      description: post.excerpt,
      type: 'article',
      publishedTime: post.date,
      images: [post.coverImage],
    },
    
    // Twitter Card
    twitter: {
      card: 'summary_large_image',
      title: post.title,
      images: [post.coverImage],
    }
  }
}

export default async function BlogPost({ params }) {
  const { slug } = await params
  const post = await getPost(slug)
  
  return <article>{post.content}</article>
}

Layout con Metadata Base

// app/layout.js
export const metadata = {
  metadataBase: new URL('https://tuosito.com'),
  
  title: {
    default: 'Il Mio Blog',
    template: '%s | Il Mio Blog'  // %s = titolo pagina
  },
  
  description: 'Blog su web development',
  
  openGraph: {
    type: 'website',
    locale: 'it_IT',
    siteName: 'Il Mio Blog',
  }
}
🎯 Title Template

Con template: '%s | Il Mio Blog', ogni pagina che definisce solo title: "Titolo" avrà automaticamente "Titolo | Il Mio Blog"!

JSON-LD per Rich Snippets

// app/blog/[slug]/page.js
export default async function BlogPost({ params }) {
  const { slug } = await params
  const post = await getPost(slug)
  
  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'Article',
    headline: post.title,
    image: post.coverImage,
    datePublished: post.date,
    author: {
      '@type': 'Person',
      name: post.author
    }
  }
  
  return (
    <>
      <script
        type="application/ld+json"
        dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
      />
      <article>{post.content}</article>
    </>
  )
}
