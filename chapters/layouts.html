Layouts e Templates

Root Layout (Obbligatorio)

// app/layout.js
export default function RootLayout({ children }) {
  return (
    <html lang="it">
      <body>
        <header>Header Globale</header>
        {children}
        <footer>Footer Globale</footer>
      </body>
    </html>
  )
}

Nested Layouts

// app/blog/layout.js
export default function BlogLayout({ children }) {
  return (
    <div className="max-w-4xl mx-auto">
      <aside>Sidebar Blog</aside>
      <main>{children}</main>
    </div>
  )
}

// Si applica a tutte le route in /blog/*

Templates vs Layouts

Layouts = Persistono tra navigazioni (non re-render)
Templates = Nuovo render ad ogni navigazione

// app/blog/template.js
export default function BlogTemplate({ children }) {
  return (
    <div className="animate-fade-in">
      {children}
    </div>
  )
}
// Si re-render ad ogni cambio pagina!
