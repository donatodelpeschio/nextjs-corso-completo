Routing File-Based

🔴 BREAKING CHANGE - Next.js 15+

Da Next.js 15, params e searchParams sono Promise!

Devi fare await:

// ❌ VECCHIO (Next.js 13-14)
export default function Page({ params }) {
  return <div>{params.slug}</div>
}

// ✅ NUOVO (Next.js 15+)
export default async function Page({ params }) {
  const { slug } = await params
  return <div>{slug}</div>
}

Route Dinamiche

app/
├── blog/
│   ├── page.js           → /blog
│   └── [slug]/
│       └── page.js       → /blog/:slug
// app/blog/[slug]/page.js - Next.js 15+
export default async function BlogPost({ params }) {
  const { slug } = await params  // ✅ await!
  
  return <h1>Post: {slug}</h1>
}
🆕 Params è Promise

Next.js 15 ha reso params asincrono per ottimizzazioni future. Sempre await!

Route Annidate

// app/shop/[category]/[product]/page.js
export default async function Product({ params }) {
  const { category, product } = await params
  
  return (
    <div>
      <h1>{category}</h1>
      <h2>{product}</h2>
    </div>
  )
}
// URL: /shop/elettronica/laptop

Search Params (Query String)

// app/search/page.js
export default async function SearchPage({ searchParams }) {
  const params = await searchParams  // ✅ await!
  const query = params.q
  
  return <h1>Cercando: {query}</h1>
}
// URL: /search?q=nextjs

Catch-All Routes

// app/docs/[...slug]/page.js
export default async function Docs({ params }) {
  const { slug } = await params
  
  return <p>Path: {slug.join('/')}</p>
}
// /docs/api/auth → slug = ['api', 'auth']

Optional Catch-All

// app/shop/[[...categories]]/page.js
export default async function Shop({ params }) {
  const { categories } = await params
  
  if (!categories) {
    return <h1>Tutti i Prodotti</h1>
  }
  
  return <h1>{categories.join(' / ')}</h1>
}
// Matcha: /shop E /shop/elettronica E /shop/elettronica/laptop
