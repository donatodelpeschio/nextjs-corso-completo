Navigazione tra Pagine

Per navigare tra pagine, usa il componente Link:

// app/page.js
import Link from 'next/link'

export default function Home() {
  return (
    <main className="p-8">
      <h1 className="text-4xl font-bold mb-4">Homepage</h1>
      <Link
        href="/about"
        className="text-cyan-400 hover:underline"
      >
        Vai alla pagina About →
      </Link>
    </main>
  )
}
⚡ Perché Link invece di <a>?

<Link> fa navigazione client-side senza ricaricare la pagina. È più veloce e mantiene lo stato dell'app. Next.js fa anche prefetching automatico dei link visibili!

Routing File-Based

🔴 IMPORTANTE - Next.js 15+ Breaking Change

Da Next.js 15, params e searchParams sono Promise!

Devi fare await prima di usarli:

// ❌ VECCHIO (Next.js 13-14)
export default function Page({ params }) {
  return <div>{params.slug}</div>
}

// ✅ NUOVO (Next.js 15+)
export default async function Page({ params }) {
  const { slug } = await params
  return <div>{slug}</div>
}

Convenzioni di Naming

Next.js usa nomi di file speciali per definire le route:

File Scopo
page.js Definisce una pagina accessibile via URL
layout.js Layout condiviso per pagine figlie
loading.js UI di caricamento (Suspense automatico)
error.js UI per gestire errori (Error Boundary)
not-found.js Pagina 404 personalizzata
route.js API endpoint (Route Handler)

Route Dinamiche (Next.js 15+)

Per creare route con parametri dinamici, usa le parentesi quadre [param]:

app/
├── blog/
│   ├── page.js           → /blog
│   └── [slug]/
│       └── page.js       → /blog/qualsiasi-cosa
// app/blog/[slug]/page.js - NEXT.JS 15+
export default async function BlogPost({ params }) {
  // ✅ IMPORTANTE: await params!
  const { slug } = await params

  return (
    <div>
      <h1>Post: {slug}</h1>
    </div>
  )
}
🆕 Perché params è una Promise?

Next.js 15 ha reso params asincrono per permettere ottimizzazioni future e supportare meglio Partial Prerendering. Devi sempre fare await params prima di accedere alle sue proprietà.

Esempi di URL che matchano:

  • /blog/primo-post → slug = "primo-post"
  • /blog/imparare-nextjs → slug = "imparare-nextjs"

Route Dinamiche Annidate

app/
└── shop/
    └── [category]/
        └── [product]/
            └── page.js   → /shop/[category]/[product]
// app/shop/[category]/[product]/page.js
export default async function Product({ params }) {
  // ✅ Unwrap della promise
  const { category, product } = await params

  return (
    <div>
      <h1>Categoria: {category}</h1>
      <h2>Prodotto: {product}</h2>
    </div>
  )
}

// URL: /shop/elettronica/laptop
// params = { category: 'elettronica', product: 'laptop' }

Search Params (Query String)

Anche searchParams è una Promise in Next.js 15+:

// app/search/page.js
export default async function SearchPage({ searchParams }) {
  // ✅ Unwrap searchParams
  const params = await searchParams
  const query = params.q
  const filter = params.filter || 'all'

  return (
    <div>
      <h1>Risultati per: {query}</h1>
      <p>Filtro: {filter}</p>
    </div>
  )
}

// URL: /search?q=nextjs&filter=tutorials

Catch-All Routes

Per catturare più segmenti di URL, usa [...slug]:

app/
└── docs/
    └── [...slug]/
        └── page.js       → /docs/a/b/c/d/e
// app/docs/[...slug]/page.js
export default async function Docs({ params }) {
  // ✅ Unwrap params
  const { slug } = await params

  return (
    <div>
      <h1>Documenti</h1>
      <p>Percorso: {slug.join('/')}</p>
    </div>
  )
}

// URL: /docs/api/users/create
// params.slug = ['api', 'users', 'create']

Optional Catch-All Routes

Con [[...slug]] (doppie parentesi) la route matcha anche senza parametri:

// app/shop/[[...categories]]/page.js
export default async function Shop({ params }) {
  const { categories } = await params

  if (!categories) {
    return <h1>Tutti i Prodotti</h1>
  }

  return <h1>Categoria: {categories.join(' / ')}</h1>
}

// Matches:
// /shop → categories = undefined
// /shop/elettronica → categories = ['elettronica']
// /shop/elettronica/laptop → categories = ['elettronica', 'laptop']

Route Groups

Organizza le route senza influenzare l'URL usando parentesi (nome):

app/
├── (marketing)/        ← Non appare nell'URL!
│   ├── about/
│   │   └── page.js    → /about (non /marketing/about)
│   └── contact/
│       └── page.js    → /contact
└── (shop)/
    └── products/
        └── page.js    → /products
🎨 Usa i Route Groups per:
  • Organizzare il codice senza cambiare URL
  • Applicare layout diversi a sezioni diverse
  • Dividere l'app in aree logiche (admin, public, ecc.)

app/

Pages Router vs App Router

Due Paradigmi Diversi

Aspetto Pages Router App Router
Introdotto Next.js 1 (2016) Next.js 13 (2022)
Stato Stable (legacy) Raccomandato (futuro)
Cartella pages/ app/
Componenti Default Client Components Server Components
Data Fetching getServerSideProps, getStaticProps async/await nei componenti
React Version React 18 React 19+ ready
Dynamic Params Sync (oggetto) Async (Promise) in 15+
💡 Il Futuro è App Router

Questo corso si concentra sull'App Router perché è il futuro di Next.js. Pages Router rimarrà supportato ma senza nuove feature. Tutte le innovazioni (Server Components, Streaming, PPR) sono solo in App Router.

Data Fetching Avanzato

Fetch in Next.js 15+

Next.js estende la Web Fetch API con caching e revalidation automatici.

Caching Automatico

// Default: cached (SSG)
const data = await fetch('https://api.example.com/posts')

// Force no cache (SSR)
const data = await fetch('https://api.example.com/posts', {
  cache: 'no-store'
})

// Revalidate ogni 60 secondi (ISR)
const data = await fetch('https://api.example.com/posts', {
  next: { revalidate: 60 }
})

Tag-based Revalidation

// Tag the cache
const data = await fetch('https://api.example.com/posts', {
  next: { tags: ['posts'] }
})

// Revalidate by tag
import { revalidateTag } from 'next/cache'
revalidateTag('posts')

export default async function sitemap() {
  const posts = await fetch('https://api.example.com/posts').then(r => r.json())

  const postUrls = posts.map(post => ({
    url: `https://tuosito.com/blog/${post.slug}`,

    {

      lastModified: new Date(),
      changeFrequency: 'yearly',
      priority: 1,
    },

  ]
}

Route Handlers (Next.js 15+)

// app/api/posts/route.js
export async function GET(request) {
  const posts = await db.posts.findMany()
  return Response.json(posts)
}

export async function POST(request) {
  const body = await request.json()
  const post = await db.posts.create({ data: body })
  return Response.json(post, { status: 201 })
}

Dynamic Route Handlers

// app/api/posts/[id]/route.js
export async function GET(request, { params }) {
  const { id } = await params  // ✅ Next.js 15+
  const post = await db.posts.findUnique({ where: { id } })

  if (!post) {
    return Response.json({ error: 'Not found' }, { status: 404 })
  }

  return Response.json(post)
}

Middleware

Cos'è il Middleware

Il middleware esegue PRIMA che una richiesta sia completata. Utile per auth, redirects, headers.

// middleware.js (root del progetto)
import { NextResponse } from 'next/server'

export function middleware(request) {
  // Check auth
  const token = request.cookies.get('token')

  if (!token && request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  return NextResponse.next()
}

export const config = {
  matcher: '/dashboard/:path*',
}

Implementazione Base

// lib/posts.js
import fs from 'fs/promises'
import path from 'path'
import matter from 'gray-matter'

const postsDirectory = path.join(process.cwd(), 'posts')

export async function getAllPosts() {
  const fileNames = await fs.readdir(postsDirectory)

  const posts = await Promise.all(
    fileNames.map(async (fileName) => {
      const slug = fileName.replace(/\.md$/, '')
      const fullPath = path.join(postsDirectory, fileName)
      const fileContents = await fs.readFile(fullPath, 'utf8')
      const { data, content } = matter(fileContents)

      return {
        slug,
        ...data,
        content,
      }
    })
  )

  return posts.sort((a, b) =>
    new Date(b.date) - new Date(a.date)
  )
}

export async function getPostBySlug(slug) {
  const posts = await getAllPosts()
  return posts.find(post => post.slug === slug)
}
// app/blog/[slug]/page.js
import { getPostBySlug, getAllPosts } from '@/lib/posts'
import { notFound } from 'next/navigation'

export async function generateMetadata({ params }) {
  const { slug } = await params
  const post = await getPostBySlug(slug)

  if (!post) return {}

  return {
    title: post.title,
    description: post.excerpt,
    alternates: {
      canonical: `/blog/${slug}`
    }
  }
}

export default async function BlogPost({ params }) {
  const { slug } = await params
  const post = await getPostBySlug(slug)

  if (!post) {
    notFound()
  }

  return (
    <article className="prose lg:prose-xl">
      <h1>{post.title}</h1>
      <time>{post.date}</time>
      <div dangerouslySetInnerHTML={{ __html: post.content }} />
    </article>
  )
}

export async function generateStaticParams() {
  const posts = await getAllPosts()
  return posts.map(post => ({ slug: post.slug }))
}
